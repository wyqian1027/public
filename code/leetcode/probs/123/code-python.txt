class Solution:
    
    def maxProfit(self, prices: List[int]) -> int:
        
        if not prices: return 0
        
        k = 3
        dp = [[0] * len(prices) for _ in range(k)]
        
        for i in range(1, k):
            low = prices[0]
            for j in range(1, len(prices)):
                low = min(low, prices[j] - dp[i-1][j-1])
                dp[i][j] = max(dp[i][j-1], prices[j] - low)
        
        return dp[-1][-1]
