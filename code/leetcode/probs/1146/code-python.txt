class SnapshotArray:

    def __init__(self, length: int):
        self.snap_id = 0
        self.store = {}
        self.arr = [0]*length
        self.length = length
        self.curMap = {}
        self.changed = True #whether changed since last snap

    def set(self, index: int, val: int) -> None:
        if self.arr[index] == val:
            return
        self.arr[index] = val
        self.curMap[index] = val
        self.changed = True

    def snap(self) -> int:
        if not self.changed:
            self.store[self.snap_id] = self.store[self.snap_id-1]
        else:
            self.store[self.snap_id] = self.curMap.copy()
            self.changed = False
        self.snap_id += 1
        return self.snap_id - 1

    def get(self, index: int, snap_id: int) -> int:
        if index in self.store[snap_id]:
            return self.store[snap_id][index]
        else:
            return 0
