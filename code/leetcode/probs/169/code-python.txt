# 1. Hash Table. O(N) Space
class Solution:
    def majorityElement(self, nums: List[int]) -> int:
        ct = collections.Counter(nums)
        n = len(nums)
        for k, v in ct.items():
            if v > n // 2: return k
            
# 2. Boyer-Moore Voting Algorithm. O(1) Space
class Solution:
    def majorityElement(self, nums: List[int]) -> int:
        count = 0
        cand = None
        for i in range(len(nums)):
            if count == 0:
                cand = nums[i]
            if cand == nums[i]:
                count += 1
            else:
                count -= 1
        return cand
