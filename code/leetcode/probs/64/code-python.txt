class Solution:
    def minPathSum(self, grid: List[List[int]]) -> int:
                      
        dp = [grid[0][0]]
        
        for c in range(1, len(grid[0])):
            dp.append(grid[0][c] + dp[-1])
        
        for r in range(1, len(grid)):
            for c in range(len(grid[0])):
                if c == 0:
                    dp[c] += grid[r][c]
                else:
                    dp[c] = min(dp[c-1], dp[c]) + grid[r][c]

        return dp[-1]
